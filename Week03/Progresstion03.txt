1. Word Frequency Counter
 ให้สร้าง function getFreqOfWords (sentence) return คำตอบเป็น key object และ ตรวจสอบ input validate
โดยจะใช้เงื่อนไขตรวจสอบ null และ undefined จากนั้นกำหนดตัวแปร words เป็นตัวพิมพ์เล็ก (toLowerCase) เพื่อให้เกิด ignore case
และใช้ split("") เพื่อแยก string เป็นคำ
let words = sentence.toLowerCase().split(" ") // ไว้ใช้แยกคำ!!

ใช้ object ในการเก็บ key (คำ) และ value (จำนวนครั้งที่เจอ) และ การวนลูป for...of เพื่อเช็คและนับความถี่ของคำ
for(let word of words) {
        if(count[word]) {
            count[word] += 1
        } else {
            count[word] = 1
        }
    }
    return count

3. ใช้ Function Expression (Arrow Function) multiply นำเลขมาคูณกัน ฝึกการเขียนฟังก์ชันในรูปแบบ Function Expression
โดยการใช้ Arrow Function (=>) แทน functionปกติ
let multiply = (a, b) => a * b 
console.log(multiply(2,2));


4. สร้าง Function Expression (Assigned Function) divide(a, b)
การประกาศฟังก์ชันโดยกำหนดให้ตัวแปร (Function Expression) เป็น divide
- การเขียนเงื่อนไข if...else ตรวจสอบค่า b จัดการกรณี error ที่เกิดจากการหารด้วยศูนย์
let divide = function(a, b) {
    if (b === 0) return "Error: divide by zero";
    return a / b;
};

5. สร้าง Arguments Object showArgs() ใช้ arguments object เพื่อเข้าถึงค่าพารามิเตอร์ทั้งหมด
จากนั้นวนลูปค่า arguments โดยไม่ต้องประกาศ parameter ในฟังก์ชัน
function showArgs() {
    for(argu of arguments) {
        console.log(argu);   
    }
}

6. สร้าง Rest Parameter sumAll(...nums)
ใช้ rest parameter (...nums) เพื่อรับค่าพารามิเตอร์แบบไม่จำกัดจำนวน, กำหนดให้ sum = 0 และใช้ for...of  ลูปรวมค่าทั้งหมด
function sumAll(...nums) {
    let sum = 0
    for(num of arguments) {
        sum += num
    }
    return sum
}


 7. เขียน function copyAndAdd(arr, num) --> Spread Operator copyAndAdd
ใช้ spread operator (...) เพื่อ copy ค่าใน array, เพิ่มค่าใหม่ลงใน array กำหนดให้ newArr = [...arr, num] 
โดยไม่กระทบกับ original array
function copyAndAdd (arr, num) {
    let newArr = [...arr, num]
    return newArr 
}
let numbers = [1, 2, 3]
let result = copyAndAdd(numbers, 4)
console.log(numbers); //[ 1, 2, 3 ]Original 
console.log(result);//  [ 1, 2, 3, 4 ] add 4 in array
