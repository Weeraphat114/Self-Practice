5.ให้สร้างฟังก์ชัน findOldest(people) เพื่อหาชื่อของคนที่มีอายุมากที่สุดในอาเรย์
เริ่มมา กำหนดคนแรกเป็นคนที่อายุมากที่สุดชั่วคราว ใช้ for-loop วนลูปผ่านทุกคน
และใช่ if เปรียบเทียบ ถ้าเกิดมีคนอายุมากกว่าคนแรกใน array ขึ้นมา ให้เปลี่ยน oldest เป็นคนนี้
let oldest = people[0]
    for (let i = 0; i < people.length; i++) {
        if (people[i].age > oldest.age) {
            oldest = people[i];
        }   
    }
